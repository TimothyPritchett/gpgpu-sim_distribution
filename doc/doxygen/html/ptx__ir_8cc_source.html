00331 
00332    // first instruction is a leader
00333    i=m_instructions.begin();
00334    leaders.push_back(*i);
00335    i++;
00336    while( i!=m_instructions.end() ) {
00337       ptx_instruction *pI = *i;
00338       if( pI->is_label() ) {
00339          leaders.push_back(pI);
00340          i = find_next_real_instruction(++i);
00341       } else {
00342          switch( pI->get_opcode() ) {
00343          case BRA_OP: case RET_OP: case EXIT_OP: case RETP_OP: case BREAK_OP: 
00344             i++;
00345             if( i != m_instructions.end() ) 
00346                leaders.push_back(*i);
00347             i = find_next_real_instruction(i);
00348             break;
00349          case CALL_OP: case CALLP_OP:
00350             if( pI->has_pred() ) {
00351                printf("GPGPU-Sim PTX: Warning found predicated call\n");
00352                i++;
00353                if( i != m_instructions.end() ) 
00354                   leaders.push_back(*i);
00355                i = find_next_real_instruction(i);
00356             } else i++;
00357             break;
00358          default:
00359             i++;
00360          }
00361       } 
00362    }
00363 
00364    if( leaders.empty() ) {
00365       printf("GPGPU-Sim PTX: Function \'%s\' has no basic blocks\n", m_name.c_str());
00366       return;
00367    }

00376       ptx_instruction *pI = *i;

00378          // found start of next basic block
00379          m_basic_blocks.back()->ptx_end = last_real_inst;

00381             m_basic_blocks.push_back( new basic_block_t(bb_id++,*find_next_real_instruction(i),NULL,0,0) );

00384          // start search for next leader

00386       }

00388       if( !pI->is_label() ) last_real_inst = pI;
00389    }

00539    for (bb_itr = m_basic_blocks.begin();bb_itr != m_basic_blocks.end(); bb_itr++) {
00540       basic_block_t *p_bb = *bb_itr; 
00541       ptx_instruction *pI = p_bb->ptx_end;
00542       if (p_bb->is_exit) //reached last basic block, no successors to link 
00543          continue;
00544       if (pI->get_opcode() == BREAK_OP) {
00545          // backup existing successor_ids for stability check
00546          std::set<int> orig_successor_ids = p_bb->successor_ids; 
00547 
00548          // erase the previous linkage with old successors 
00549          for(std::set<int>::iterator succ_ids = p_bb->successor_ids.begin(); succ_ids != p_bb->successor_ids.end(); ++succ_ids) {
00550             basic_block_t *successor_bb = m_basic_blocks[*succ_ids];
00551             successor_bb->predecessor_ids.erase(p_bb->bb_id); 
00552          }
00553          p_bb->successor_ids.clear(); 
00554 
00555          //find successor and link that basic_block to this one
00556          //successor of a break is set by an preceeding breakaddr instruction 
00557          operand_info *target = find_break_target(pI); 
00558          unsigned addr = labels[ target->name() ];
00559          ptx_instruction *target_pI = m_instr_mem[addr];
00560          basic_block_t *target_bb = target_pI->get_bb();
00561          p_bb->successor_ids.insert(target_bb->bb_id);
00562          target_bb->predecessor_ids.insert(p_bb->bb_id);
00563 
00564          if (pI->has_pred()) {
00565             // predicated break - add link to next basic block
00566             unsigned next_addr = pI->get_m_instr_mem_index() + pI->inst_size();
00567             basic_block_t *next_bb = m_instr_mem[next_addr]->get_bb();
00568             p_bb->successor_ids.insert(next_bb->bb_id);
00569             next_bb->predecessor_ids.insert(p_bb->bb_id);
00570          }
00571 
00572          modified = modified || (orig_successor_ids != p_bb->successor_ids); 
00573       }
00574    }
00575 
00576    return modified; 
00577 }

00617    std::vector<basic_block_t*>::iterator bb_itr = m_basic_blocks.begin();
00618    (*bb_itr)->dominator_ids.insert((*bb_itr)->bb_id);  // the only dominator of the entry block is the entry
00619    //copy all basic blocks to all dominator lists EXCEPT for the entry block
00620    for (++bb_itr;bb_itr != m_basic_blocks.end(); bb_itr++) { 
00621       for (unsigned i = 0; i < m_basic_blocks.size(); i++)

00624    bool change = true;

00626       change = false;
00627       for ( int h = 1/*skip entry*/; h < m_basic_blocks.size(); ++h ) {
00628          assert( m_basic_blocks[h]->bb_id == (unsigned)h );
00629          std::set<int> T;
00630          for (unsigned i=0;i< m_basic_blocks.size();i++)

00633             intersect(T, m_basic_blocks[*s]->dominator_ids);

00636             change = true;
00637             m_basic_blocks[h]->dominator_ids = T;
00638          }
00639       }
00640    }
00641    //clean the basic block of dominators of it has no predecessors -- except for entry block
00642    bb_itr = m_basic_blocks.begin();
00643    for (++bb_itr;bb_itr != m_basic_blocks.end(); bb_itr++) {
00644       if ((*bb_itr)->predecessor_ids.empty())

00221 
00222     switch( space.get_type() ) {
00223     case shared_space: {
00224         unsigned subwarp_size = m_config->warp_size / m_config->mem_warp_parts;
00225         unsigned total_accesses=0;
00226         for( unsigned subwarp=0; subwarp <  m_config->mem_warp_parts; subwarp++ ) {
00227 
00228             // data structures used per part warp 
00229             std::map<unsigned,std::map<new_addr_type,unsigned> > bank_accs; // bank -> word address -> access count
00230 
00231             // step 1: compute accesses to words in banks
00232             for( unsigned thread=subwarp*subwarp_size; thread < (subwarp+1)*subwarp_size; thread++ ) {
00233                 if( !active(thread) ) 
00234                     continue;

00236                 //FIXME: deferred allocation of shared memory should not accumulate across kernel launches
00237                 //assert( addr < m_config->gpgpu_shmem_size ); 
00238                 unsigned bank = m_config->shmem_bank_func(addr);
00239                 new_addr_type word = line_size_based_tag_func(addr,m_config->WORD_SIZE);
00240                 bank_accs[bank][word]++;
00241             }
00242 
00243             if (m_config->shmem_limited_broadcast) {
00244                 // step 2: look for and select a broadcast bank/word if one occurs
00245                 bool broadcast_detected = false;
00246                 new_addr_type broadcast_word=(new_addr_type)-1;
00247                 unsigned broadcast_bank=(unsigned)-1;
00248                 std::map<unsigned,std::map<new_addr_type,unsigned> >::iterator b;
00249                 for( b=bank_accs.begin(); b != bank_accs.end(); b++ ) {
00250                     unsigned bank = b->first;
00251                     std::map<new_addr_type,unsigned> &access_set = b->second;
00252                     std::map<new_addr_type,unsigned>::iterator w;
00253                     for( w=access_set.begin(); w != access_set.end(); ++w ) {
00254                         if( w->second > 1 ) {
00255                             // found a broadcast
00256                             broadcast_detected=true;
00257                             broadcast_bank=bank;
00258                             broadcast_word=w->first;
00259                             break;
00260                         }
00261                     }
00262                     if( broadcast_detected ) 
00263                         break;
00264                 }
00265             
00266                 // step 3: figure out max bank accesses performed, taking account of broadcast case
00267                 unsigned max_bank_accesses=0;
00268                 for( b=bank_accs.begin(); b != bank_accs.end(); b++ ) {
00269                     unsigned bank_accesses=0;
00270                     std::map<new_addr_type,unsigned> &access_set = b->second;
00271                     std::map<new_addr_type,unsigned>::iterator w;
00272                     for( w=access_set.begin(); w != access_set.end(); ++w ) 
00273                         bank_accesses += w->second;
00274                     if( broadcast_detected && broadcast_bank == b->first ) {
00275                         for( w=access_set.begin(); w != access_set.end(); ++w ) {
00276                             if( w->first == broadcast_word ) {
00277                                 unsigned n = w->second;
00278                                 assert(n > 1); // or this wasn't a broadcast
00279                                 assert(bank_accesses >= (n-1));
00280                                 bank_accesses -= (n-1);
00281                                 break;
00282                             }
00283                         }
00284                     }
00285                     if( bank_accesses > max_bank_accesses ) 
00286                         max_bank_accesses = bank_accesses;
00287                 }
00288 
00289                 // step 4: accumulate
00290                 total_accesses+= max_bank_accesses;
00291             } else {
00292                 // step 2: look for the bank with the maximum number of access to different words 
00293                 unsigned max_bank_accesses=0;
00294                 std::map<unsigned,std::map<new_addr_type,unsigned> >::iterator b;
00295                 for( b=bank_accs.begin(); b != bank_accs.end(); b++ ) {
00296                     max_bank_accesses = std::max(max_bank_accesses, (unsigned)b->second.size());
00297                 }
00298 
00299                 // step 3: accumulate
00300                 total_accesses+= max_bank_accesses;
00301             }
00302         }
00303         assert( total_accesses > 0 && total_accesses <= m_config->warp_size );
00304         cycles = total_accesses; // shared memory conflicts modeled as larger initiation interval 
00305         ptx_file_line_stats_add_smem_bank_conflict( pc, total_accesses );
00306         break;
00307     }
00308 
00309     case tex_space: 
00310         cache_block_size = m_config->gpgpu_cache_texl1_linesize;
00311         break;
00312     case const_space:  case param_space_kernel:
00313         cache_block_size = m_config->gpgpu_cache_constl1_linesize; 
00314         break;
00315 
00316     case global_space: case local_space: case param_space_local:
00317         if( m_config->gpgpu_coalesce_arch == 13 ) {
00318            if(isatomic())
00319                memory_coalescing_arch_13_atomic(is_write, access_type);
00320            else
00321                memory_coalescing_arch_13(is_write, access_type);
00322         } else abort();
00323 
00324         break;
00325 
00326     default:
00327         abort();
00328     }
00329 
00330     if( cache_block_size ) {
00331         assert( m_accessq.empty() );
00332         mem_access_byte_mask_t byte_mask; 
00333         std::map<new_addr_type,active_mask_t> accesses; // block address -> set of thread offsets in warp
00334         std::map<new_addr_type,active_mask_t>::iterator a;
00335         for( unsigned thread=0; thread < m_config->warp_size; thread++ ) {
00336             if( !active(thread) ) 
00337                 continue;
00338             new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[0];
00339             unsigned block_address = line_size_based_tag_func(addr,cache_block_size);
00340             accesses[block_address].set(thread);
00341             unsigned idx = addr-block_address; 
00342             for( unsigned i=0; i < data_size; i++ ) 
00343                 byte_mask.set(idx+i);
00344         }
00345         for( a=accesses.begin(); a != accesses.end(); ++a ) 
00346             m_accessq.push_back( mem_access_t(access_type,a->first,cache_block_size,is_write,a->second,byte_mask) );
00347     }
00348 
00349     if ( space.get_type() == global_space ) {
00350         ptx_file_line_stats_add_uncoalesced_gmem( pc, m_accessq.size() - starting_queue_size );
00351     }
00352     m_mem_accesses_created=true;
00353 }

00355 void warp_inst_t::memory_coalescing_arch_13( bool is_write, mem_access_type access_type )
00356 {
00357     // see the CUDA manual where it discusses coalescing rules before reading this
00358     unsigned segment_size = 0;
00359     unsigned warp_parts = m_config->mem_warp_parts;
00360     switch( data_size ) {
00361     case 1: segment_size = 32; break;
00362     case 2: segment_size = 64; break;
00363     case 4: case 8: case 16: segment_size = 128; break;
00364     }
00365     unsigned subwarp_size = m_config->warp_size / warp_parts;
00366 
00367     for( unsigned subwarp=0; subwarp <  warp_parts; subwarp++ ) {
00368         std::map<new_addr_type,transaction_info> subwarp_transactions;
00369 
00370         // step 1: find all transactions generated by this subwarp
00371         for( unsigned thread=subwarp*subwarp_size; thread<subwarp_size*(subwarp+1); thread++ ) {
00372             if( !active(thread) )
00373                 continue;
00374 
00375             unsigned data_size_coales = data_size;
00376             unsigned num_accesses = 1;
00377 
00378             if( space.get_type() == local_space || space.get_type() == param_space_local ) {
00379                // Local memory accesses >4B were split into 4B chunks
00380                if(data_size >= 4) {
00381                   data_size_coales = 4;
00382                   num_accesses = data_size/4;
00383                }
00384                // Otherwise keep the same data_size for sub-4B access to local memory
00385             }
00386 
00387 
00388             assert(num_accesses <= MAX_ACCESSES_PER_INSN_PER_THREAD);
00389 
00390             for(unsigned access=0; access<num_accesses; access++) {
00391                 new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[access];
00392                 unsigned block_address = line_size_based_tag_func(addr,segment_size);
00393                 unsigned chunk = (addr&127)/32; // which 32-byte chunk within in a 128-byte chunk does this thread access?
00394                 transaction_info &info = subwarp_transactions[block_address];
00395 
00396                 // can only write to one segment
00397                 assert(block_address == line_size_based_tag_func(addr+data_size_coales-1,segment_size));
00398 
00399                 info.chunks.set(chunk);
00400                 info.active.set(thread);
00401                 unsigned idx = (addr&127);
00402                 for( unsigned i=0; i < data_size_coales; i++ )
00403                     info.bytes.set(idx+i);
00404             }
00405         }
00406 
00407         // step 2: reduce each transaction size, if possible
00408         std::map< new_addr_type, transaction_info >::iterator t;
00409         for( t=subwarp_transactions.begin(); t !=subwarp_transactions.end(); t++ ) {
00410             new_addr_type addr = t->first;
00411             const transaction_info &info = t->second;
00412 
00413             memory_coalescing_arch_13_reduce_and_send(is_write, access_type, info, addr, segment_size);
00414 
00415         }
00416     }
00417 }

00420 {
00421 
00422    assert(space.get_type() == global_space); // Atomics allowed only for global memory
00423 
00424    // see the CUDA manual where it discusses coalescing rules before reading this
00425    unsigned segment_size = 0;
00426    unsigned warp_parts = 2;
00427    switch( data_size ) {
00428    case 1: segment_size = 32; break;
00429    case 2: segment_size = 64; break;

00436

00438        for( unsigned thread=subwarp*subwarp_size; thread<subwarp_size*(subwarp+1); thread++ ) {
00439            if( !active(thread) )
00440                continue;

00476

00524            if(q[1]) h.set(1);

00528            if(q[3]) h.set(1);

00760 void simt_stack::update( simt_mask_t &thread_done, addr_vector_t &next_pc, address_type recvg_pc, op_type next_inst_op,unsigned next_inst_size, address_type next_inst_pc )
00761 {
00762     assert(m_stack.size() > 0);
00763 
00764     assert( next_pc.size() == m_warp_size );
00765 
00766     simt_mask_t  top_active_mask = m_stack.back().m_active_mask;
00767     address_type top_recvg_pc = m_stack.back().m_recvg_pc;
00768     address_type top_pc = m_stack.back().m_pc; // the pc of the instruction just executed
00769     stack_entry_type top_type = m_stack.back().m_type;
00770     assert(top_pc==next_inst_pc);
00771     assert(top_active_mask.any());
00772 
00773     const address_type null_pc = -1;
00774     bool warp_diverged = false;
00775     address_type new_recvg_pc = null_pc;
00776     unsigned num_divergent_paths=0;
00777 
00778     std::map<address_type,simt_mask_t> divergent_paths;
00779     while (top_active_mask.any()) {
00780 
00781         // extract a group of threads with the same next PC among the active threads in the warp
00782         address_type tmp_next_pc = null_pc;
00783         simt_mask_t tmp_active_mask;
00784         for (int i = m_warp_size - 1; i >= 0; i--) {
00785             if ( top_active_mask.test(i) ) { // is this thread active?
00786                 if (thread_done.test(i)) {
00787                     top_active_mask.reset(i); // remove completed thread from active mask
00788                 } else if (tmp_next_pc == null_pc) {
00789                     tmp_next_pc = next_pc[i];
00790                     tmp_active_mask.set(i);
00791                     top_active_mask.reset(i);
00792                 } else if (tmp_next_pc == next_pc[i]) {
00793                     tmp_active_mask.set(i);
00794                     top_active_mask.reset(i);
00795                 }
00796             }
00797         }
00798 
00799         if(tmp_next_pc == null_pc) {
00800             assert(!top_active_mask.any()); // all threads done
00801             continue;
00802         }
00803 
00804         divergent_paths[tmp_next_pc]=tmp_active_mask;
00805         num_divergent_paths++;
00806     }
00807 
00808 
00809     address_type not_taken_pc = next_inst_pc+next_inst_size;
00810     assert(num_divergent_paths<=2);
00811     for(unsigned i=0; i<num_divergent_paths; i++){
00812         address_type tmp_next_pc = null_pc;
00813         simt_mask_t tmp_active_mask;
00814         tmp_active_mask.reset();
00815         if(divergent_paths.find(not_taken_pc)!=divergent_paths.end()){
00816             assert(i==0);
00817             tmp_next_pc=not_taken_pc;
00818             tmp_active_mask=divergent_paths[tmp_next_pc];
00819             divergent_paths.erase(tmp_next_pc);
00820         }else{
00821             std::map<address_type,simt_mask_t>:: iterator it=divergent_paths.begin();
00822             tmp_next_pc=it->first;
00823             tmp_active_mask=divergent_paths[tmp_next_pc];
00824             divergent_paths.erase(tmp_next_pc);
00825         }
00826 
00827         // HANDLE THE SPECIAL CASES FIRST
00828         if (next_inst_op== CALL_OPS){
00829             // Since call is not a divergent instruction, all threads should have executed a call instruction
00830             assert(num_divergent_paths == 1);
00831 
00832             simt_stack_entry new_stack_entry;
00833             new_stack_entry.m_pc = tmp_next_pc;
00834             new_stack_entry.m_active_mask = tmp_active_mask;
00835             new_stack_entry.m_branch_div_cycle = gpu_sim_cycle+gpu_tot_sim_cycle;
00836             new_stack_entry.m_type = STACK_ENTRY_TYPE_CALL;
00837             m_stack.push_back(new_stack_entry);
00838             return;
00839         }else if(next_inst_op == RET_OPS && top_type==STACK_ENTRY_TYPE_CALL){
00840             // pop the CALL Entry
00841             assert(num_divergent_paths == 1);
00842             m_stack.pop_back();
00843 
00844             assert(m_stack.size() > 0);
00845             m_stack.back().m_pc=tmp_next_pc;// set the PC of the stack top entry to return PC from  the call stack;
00846             // Check if the New top of the stack is reconverging
00847             if (tmp_next_pc == m_stack.back().m_recvg_pc && m_stack.back().m_type!=STACK_ENTRY_TYPE_CALL){
00848                 assert(m_stack.back().m_type==STACK_ENTRY_TYPE_NORMAL);
00849                 m_stack.pop_back();
00850             }
00851             return;
00852         }
00853 
00854         // discard the new entry if its PC matches with reconvergence PC
00855         // that automatically reconverges the entry
00856         // If the top stack entry is CALL, dont reconverge.
00857         if (tmp_next_pc == top_recvg_pc && (top_type != STACK_ENTRY_TYPE_CALL)) continue;
00858 
00859         // this new entry is not converging
00860         // if this entry does not include thread from the warp, divergence occurs
00861         if ((num_divergent_paths>1) && !warp_diverged ) {
00862             warp_diverged = true;
00863             // modify the existing top entry into a reconvergence entry in the pdom stack
00864             new_recvg_pc = recvg_pc;
00865             if (new_recvg_pc != top_recvg_pc) {
00866                 m_stack.back().m_pc = new_recvg_pc;
00867                 m_stack.back().m_branch_div_cycle = gpu_sim_cycle+gpu_tot_sim_cycle;
00868 
00869                 m_stack.push_back(simt_stack_entry());
00870             }
00871         }
00872 
00873         // discard the new entry if its PC matches with reconvergence PC
00874         if (warp_diverged && tmp_next_pc == new_recvg_pc) continue;
00875 
00876         // update the current top of pdom stack
00877         m_stack.back().m_pc = tmp_next_pc;
00878         m_stack.back().m_active_mask = tmp_active_mask;
00879         if (warp_diverged) {
00880             m_stack.back().m_calldepth = 0;
00881             m_stack.back().m_recvg_pc = new_recvg_pc;
00882         } else {
00883             m_stack.back().m_recvg_pc = top_recvg_pc;
00884         }
00885 
00886         m_stack.push_back(simt_stack_entry());
00887     }
00888     assert(m_stack.size() > 0);
00889     m_stack.pop_back();
00890 
00891 
00892     if (warp_diverged) {
00893         ptx_file_line_stats_add_warp_divergence(top_pc, 1); 
00894     }
00895 }

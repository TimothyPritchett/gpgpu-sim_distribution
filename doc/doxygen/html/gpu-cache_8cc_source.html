00050 unsigned l1d_cache_config::set_index(new_addr_type addr) const{
00051     unsigned set_index = m_nset; // Default to linear set index function
00052     unsigned lower_xor = 0;
00053     unsigned upper_xor = 0;
00054 
00055     switch(m_set_index_function){
00056     case FERMI_HASH_SET_FUNCTION:
00057         /*
00058         * Set Indexing function from "A Detailed GPU Cache Model Based on Reuse Distance Theory"
00059         * Cedric Nugteren et al.
00060         * ISCA 2014
00061         */
00062         if(m_nset == 32 || m_nset == 64){
00063             // Lower xor value is bits 7-11
00064             lower_xor = (addr >> m_line_sz_log2) & 0x1F;
00065 
00066             // Upper xor value is bits 13, 14, 15, 17, and 19
00067             upper_xor  = (addr & 0xE000)  >> 13; // Bits 13, 14, 15
00068             upper_xor |= (addr & 0x20000) >> 14; // Bit 17
00069             upper_xor |= (addr & 0x80000) >> 15; // Bit 19
00070 
00071             set_index = (lower_xor ^ upper_xor);
00072 
00073             // 48KB cache prepends the set_index with bit 12
00074             if(m_nset == 64)
00075                 set_index |= (addr & 0x1000) >> 7;
00076 
00077         }else{ /* Else incorrect number of sets for the hashing function */
00078             assert("\nGPGPU-Sim cache configuration error: The number of sets should be "
00079                     "32 or 64 for the hashing set index function.\n" && 0);
00080         }
00081         break;
00082 
00083     case CUSTOM_SET_FUNCTION:
00084         /* No custom set function implemented */
00085         break;
00086 
00087     case LINEAR_SET_FUNCTION:
00088         set_index = (addr >> m_line_sz_log2) & (m_nset-1);
00089         break;
00090     }
00091 
00092     // Linear function selected or custom set index function not implemented
00093     assert((set_index < m_nset) && "\nError: Set index out of bounds. This is caused by "
00094             "an incorrect or unimplemented custom set index function.\n");
00095 
00096     return set_index;
00097 }

00158 enum cache_request_status tag_array::probe( new_addr_type addr, unsigned &idx ) const {
00159     //assert( m_config.m_write_policy == READ_ONLY );
00160     unsigned set_index = m_config.set_index(addr);
00161     new_addr_type tag = m_config.tag(addr);
00162 
00163     unsigned invalid_line = (unsigned)-1;
00164     unsigned valid_line = (unsigned)-1;
00165     unsigned valid_timestamp = (unsigned)-1;
00166 
00167     bool all_reserved = true;
00168 
00169     // check for hit or pending hit
00170     for (unsigned way=0; way<m_config.m_assoc; way++) {
00171         unsigned index = set_index*m_config.m_assoc+way;
00172         cache_block_t *line = &m_lines[index];
00173         if (line->m_tag == tag) {
00174             if ( line->m_status == RESERVED ) {
00175                 idx = index;
00176                 return HIT_RESERVED;
00177             } else if ( line->m_status == VALID ) {
00178                 idx = index;
00179                 return HIT;
00180             } else if ( line->m_status == MODIFIED ) {
00181                 idx = index;
00182                 return HIT;
00183             } else {
00184                 assert( line->m_status == INVALID );
00185             }
00186         }
00187         if (line->m_status != RESERVED) {
00188             all_reserved = false;
00189             if (line->m_status == INVALID) {
00190                 invalid_line = index;
00191             } else {
00192                 // valid line : keep track of most appropriate replacement candidate
00193                 if ( m_config.m_replacement_policy == LRU ) {
00194                     if ( line->m_last_access_time < valid_timestamp ) {
00195                         valid_timestamp = line->m_last_access_time;
00196                         valid_line = index;
00197                     }
00198                 } else if ( m_config.m_replacement_policy == FIFO ) {
00199                     if ( line->m_alloc_time < valid_timestamp ) {
00200                         valid_timestamp = line->m_alloc_time;
00201                         valid_line = index;
00202                     }
00203                 }
00204             }
00205         }
00206     }
00207     if ( all_reserved ) {
00208         assert( m_config.m_alloc_policy == ON_MISS ); 
00209         return RESERVATION_FAIL; // miss and not enough space in cache to allocate on miss
00210     }
00211 
00212     if ( invalid_line != (unsigned)-1 ) {
00213         idx = invalid_line;
00214     } else if ( valid_line != (unsigned)-1) {
00215         idx = valid_line;
00216     } else abort(); // if an unreserved block exists, it is either invalid or replaceable 
00217 
00218     return MISS;
00219 }

00623 void baseline_cache::bandwidth_management::use_data_port(mem_fetch *mf, enum cache_request_status outcome, const std::list<cache_event> &events)
00624 {
00625     unsigned data_size = mf->get_data_size(); 
00626     unsigned port_width = m_config.m_data_port_width; 
00627     switch (outcome) {
00628     case HIT: {
00629         unsigned data_cycles = data_size / port_width + ((data_size % port_width > 0)? 1 : 0); 
00630         m_data_port_occupied_cycles += data_cycles; 
00631         } break; 
00632     case HIT_RESERVED: 
00633     case MISS: {
00634         // the data array is accessed to read out the entire line for write-back 
00635         if (was_writeback_sent(events)) {
00636             unsigned data_cycles = m_config.m_line_sz / port_width; 
00637             m_data_port_occupied_cycles += data_cycles; 
00638         }
00639         } break; 
00640     case RESERVATION_FAIL: 
00641         // Does not consume any port bandwidth 
00642         break; 
00643     default: 
00644         assert(0); 
00645         break; 
00646     } 
00647 }

01117     unsigned time, std::list<cache_event> &events )
01118 {
01119     if ( m_fragment_fifo.full() || m_request_fifo.full() || m_rob.full() )
01120         return RESERVATION_FAIL;
01121 
01122     assert( mf->get_data_size() <= m_config.get_line_sz());

01144         cache_status = HIT_RESERVED;

01150 void tex_cache::cycle(){
01151     // send next request to lower level of memory
01152     if ( !m_request_fifo.empty() ) {
01153         mem_fetch *mf = m_request_fifo.peek();
01154         if ( !m_memport->full(mf->get_ctrl_size(),false) ) {
01155             m_request_fifo.pop();
01156             m_memport->push(mf);
01157         }
01158     }
01159     // read ready lines from cache
01160     if ( !m_fragment_fifo.empty() && !m_result_fifo.full() ) {
01161         const fragment_entry &e = m_fragment_fifo.peek();
01162         if ( e.m_miss ) {
01163             // check head of reorder buffer to see if data is back from memory
01164             unsigned rob_index = m_rob.next_pop_index();
01165             const rob_entry &r = m_rob.peek(rob_index);
01166             assert( r.m_request == e.m_request );
01167             assert( r.m_block_addr == m_config.block_addr(e.m_request->get_addr()) );
01168             if ( r.m_ready ) {
01169                 assert( r.m_index == e.m_cache_index );

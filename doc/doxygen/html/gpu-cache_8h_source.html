00181         }

00184         case 'A': m_mshr_type = ASSOC; break;
00185         default: exit_parse_error();

00188         m_nset_log2 = LOGB2(m_nset);

00190

00192         case 'W': m_write_alloc_policy = WRITE_ALLOCATE; break;
00193         case 'N': m_write_alloc_policy = NO_WRITE_ALLOCATE; break;
00194         default: exit_parse_error();
00195         }

00198         if (m_alloc_policy == ON_FILL and m_write_policy == WRITE_BACK) {
00199             // A writeback cache with allocate-on-fill policy will inevitably lead to deadlock:

00204             // incoming cache-fill.  The stall may propagate through the memory

00209         }
